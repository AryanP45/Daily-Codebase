# [1588. Sum of All Odd Length Subarrays](https://leetcode.com/problems/sum-of-all-odd-length-subarrays/)
> Version 1
```c++
 class Solution {
public:
    int sumOddLengthSubarrays(vector<int>& arr) {
        int sum=0;
    //    if(arr.size() == 1) arr[0];
        for(int i=1;i<=arr.size();i+=2){
            for(int j=0;i+j <= arr.size();j++){

                for(int k=j;k<j+i;k++)
                    sum += arr[k];
            }
            
        }
        return sum;
    }
};
```
