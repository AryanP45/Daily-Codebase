# [1716. Calculate Money in Leetcode Bank](https://leetcode.com/problems/calculate-money-in-leetcode-bank/)
> Version 1
```java
 class Solution {
    public int totalMoney(int n) {
        int money =0;
        int mon = 0;
        int cur = 1;
        for(int i=1;i<=n;i++){
            if( i % 7 != 1){
                money += cur++;
            }else{
                cur = ++mon;
                money += cur++;
                System.out.println(" day : "+i+" cur :"+mon);
            }
        }
        return money;
    }
}
```

> Version 2 (Alternate approach/Optimized)
```java
class Solution {
    public int totalMoney(int days) {

        // solution with arithmetic progression
        float money = 0;
        float a = 0;
        float d = 1;
        
        int weeks = days / 7;

        for(int i=1;i<=weeks ; i++){
            a++;
            float n = 7;
            // float sum = ((n/2) * ((2*a) + ((n-1)*d)));
            money += weekSum(a,d,n);
            System.out.println(money);
        }

        // rest of days
        int rem = days % 7;
        System.out.println("remaining : "+rem+" a : "+a);
        if(rem != 0){
            money += weekSum( a==0?1:++a ,d,rem);
        }

        return  (int)money;
    }

    public float weekSum(float a,float d,float n){
        return ((n/2) * ((2*a) + ((n-1)*d)));
    }
}
```
