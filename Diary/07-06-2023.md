# [1492. The kth Factor of n](https://leetcode.com/problems/the-kth-factor-of-n/)
> Version 1
```c++
 class Solution {
public:
    int kthFactor(int n, int k) {
        int count=0;
        //counting possible factors
        for(int i=1;i<=n;i++){
            if(n % i == 0) count++;
            // returning kth factor
            if(count==k) return i;
        }
        //if factors less than k then returning -1
        if(count <  k) return -1;
        return 0;
    }
};
```
