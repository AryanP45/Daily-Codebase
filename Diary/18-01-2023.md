# [35. Search Insert Position](https://leetcode.com/problems/search-insert-position/)

```c++
class Solution {
public:
    int searchInsert(vector<int>& nums, int target) {
        int ans=0;
        // bool flag=false;
        for(int i=0;i<nums.size();i++){
            if(nums[i]<=target){
                ans=i+1;
                if(nums[i]==target)
                {
                    ans=i;
                    break;
                }
            }
        }
   

        return ans;
    }
};
```
