 ## Final version

 ```c++
 /**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* list1, ListNode* list2) {
            ListNode* ans;
            ListNode* ptr;
            ListNode* ctr;
            vector<int> vec;
            
            // inserting all elements in vector then apply sorting
            while(list1){
                vec.push_back(list1->val);
                list1 = list1->next;
            }
            while(list2){
                vec.push_back(list2->val);
                list2 = list2->next;
            }
            sort(vec.begin(),vec.end()); 

            // making new list from sorted vector of elements
            // 1. using loop to insert elements directly
            // 2. using append function to insert elements one by one at the tail
            for(int i=0;i<vec.size();i++){
                ptr = new ListNode();
                ptr->val = vec[i];
                if(i == 0){
                    ans = ptr;
                    ctr = ans;
                }
                else{
                    ctr->next = ptr;
                    ctr = ctr->next;
                }
                // append(&ans,vec[i]);
            }
            
            return ans;
    }
    
    // function to append element at tail of list given ref to list and value to add
    void append(ListNode** headref,int val){
        ListNode* new_node = new ListNode();
        ListNode* last= *headref;
        new_node->val=val;
        new_node->next = NULL;
        if(*headref == NULL){
            *headref = new_node;
            return;
        }

        while(last->next != NULL){
            last = last->next;
        }

        last->next = new_node;
        return;
    }
};
```
