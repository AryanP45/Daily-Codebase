# [2325. Decode the Message](https://leetcode.com/problems/decode-the-message/)
```c++
 class Solution {
public:
    string decodeMessage(string key, string message) {
        vector<bool> flag(26,false);
        char map[26];
        int curAlp=97;
        int c=0;
        for(int i=0;i<key.length();i++){

            if(key[i] != ' '){
                if(flag[key[i]-97] == false){
                    map[key[i]-97] = (char)curAlp;
                    flag[key[i]-97] = true;
                    curAlp++;
                    c++;
                }
            }
            // else{
            //     map[27][1] = key[i];
            //     map[27][0] = " ";
            // }
        }

        string ans="";
        for(int j=0;j<message.length();j++){
            if(message[j] == ' '){
                ans += " ";
            }else{
                int index= message[j]-97;
                cout<<index;
                ans += map[index];
            }
        }

        return ans;
    }
};
```
